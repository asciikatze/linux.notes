# APACHE2 webserver -----------------------------------------------------------
# install
sudo apt install apache2 apache2-utils

# make the directories
sudo mkdir -p /var/www/{mywebpage}/logs
sudo mkdir -p /var/www/{mywebpage}/public_html

# move the website files into the public_html folder

# set file ownership
sudo chown -R www-data:www-data /var/www/{mywebpage}/public_html

# make the configuration for the site, use 000-default.conf as a template
sudo nano /etc/apache2/sites-available/{mywebpage}.conf
<VirtualHost *:80>
	ServerAdmin webmaster@localhost
	DocumentRoot /var/www/{mywebpage}/public_html
	ErrorLog /var/www/{mywebpage}/logs/error.log
	CustomLog /var/www/{mywebpage}/logs/access.log combined
</VirtualHost>

sudo a2dissite 000-default.conf  #disable default site
sudo a2ensite {mywebpage}.conf   #enable the site

sudo systemctl reload apache2

# add custom error pages
file: /var/www/{mywebpage}/public_html/.htaccess
  ErrorDocument 404 /404.html
  ErrorDocument 405 /405.html

# file hosting page customization
file: /etc/apache2/apache2.conf
<Directory /var/www/{mywebpage}/public_html>
	AllowOverride All
	ServerSignature off
</Directory>

file: /var/www/{mywebpage}/public_html/{folder}/.htaccess
  IndexOptions FancyIndexing
  IndexOptions SuppressIcon
  IndexOptions SuppressLastModified
  IndexOptions SuppressDescription
# if you dont want fileshare with indexed pages, add this:
  Options -Indexes
# to .htaccess file OR /etc/apache2/apache2.conf file

# password protection on site
sudo apt install apache2-utils
sudo htpasswd -c /etc/apache2/.htpasswd {loginname}
file: /var/www/{mywebpage}/public_html/.htaccess
  AuthType Basic
  AuthName "{messagethatshowsup}"
  AuthUserFile /etc/apache2/.htpasswd
  Require valid-user
