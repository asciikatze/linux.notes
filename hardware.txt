# wipe a drive
# make a file for your script
file: wipe-sdb.sh
EOF>
  #!/bin/bash	
  while true
  do
    ((a=a+1))
    dd if=/dev/urandom of=/dev/sdb status=progress	
    echo Looped $a times
  done
<EOF
chmod +x wipe-sdb.sh
sudo ./wipe-sdb.sh
# let it run about 5 times for best result
