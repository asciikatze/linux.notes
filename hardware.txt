# sdb usually means the second connected hdd device
# sdb1 and sdb2 refrence the partitions on the disk sdb
# in these examples i'll use sdX where X represents the drive (sdX can be a valid drive)
# NOT always gonna be sdX something! sdX is just the common SCSI disk designation
# nvme and other drives can use different designations for the devices!

# partitioning and formatting
sudo cfdisk /dev/sdX  #partition editing tui tool
sudo fdisk /dev/sdX   #follow the menu

sudo mkfs.ext4 /dev/sdX        #format ext4
sudo mkfs.ntfs /dev/sdX        #format ntfs
sudo mkfs.vfat -F 32 /dev/sdX  #format fat32

sudo wipefs /dev/sdX*    #prints info about all partitions on sdX
sudo wipefs -a /dev/sdX  #erase all filesystem raid pratition signatures

# mount HDD
sudo mkdir /mnt/disk2           #make a directory to be mounted to
sudo mount /dev/sdX /mnt/disk2  #in this example the hdd will be mounted under the folder: disk2/
sudo umount -l /dev/sdX         #unmount
sudo mount | grep sdX           #list down all the mounted drives and partitions.

# add disk to fstab to mount it on boot
# help: https://wiki.archlinux.org/title/Fstab
lsblk -f  #shows devices with UUIDs
file: /etc/fstab
  UUID=xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxx /mnt/disk2 ext4 defaults,nofail 0 2  #nofail means it will boot if hdd is not connected
sudo systemctl daemon-reload  OR  sudo mount -a  #if changes happened and need to reload the fstab file
sudo dmesg                                       #if there are issues look here

# disk health
sudo badblocks -sv /dev/sdX  #-s displays progress, -v verbose mode
sudo badblocks /dev/sdX      #non-destructive read-only test
sudo badblocks -n /dev/sdX   #non-destructive read-write test
sudo badblocks -w /dev/sdX   #destructive write test
sudo badblocks -o {file.txt} -w /dev/sdX  #destructive and output found blocks to a file

# wipe a drive
#make a script file: wipe-sdX.sh
  #!/bin/bash	
  while true
  do
    ((a=a+1))
    dd if=/dev/urandom of=/dev/sdX status=progress	
    echo Looped $a times
  done
#let it run about 5 times for best result

# random data from the Linux kernel random number generator function:
/dev/random   #produces high-quality randomness, can halt process execution when it runs out of entropy
/dev/urandom  #not care about entropy, may not produce true randomness
/dev/null     #everything you put here gets discarded
