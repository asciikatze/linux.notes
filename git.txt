# ssh
ssh-keygen -t ed25519 -C "{text}"  #create keypair
ssh-add ~/.ssh/{private-keyfile}   #add pirvate key to the ssh agent
#GO TO github.com and set the public key on github.com
ssh -T git@github.com              #test the connection

file: ~/.ssh/config
	Host github-personal
	  Hostname github.com
	  User git
	  IdentityFile ~/.ssh/{private-keyfile}
	  IdentitiesOnly yes

# set up a local repository
mkdir /home/user1/git
#navigate to that directory
git clone https://github.com/username/repository.git
#set your identity
git config user.name  AND  git config user.email  #shows what is being used
git config user.name {name}                 #set name
git config user.email {dummy-github-email}  #set email
#set the remote repository
git remote -v  #check the link to the remote repo
git remote add origin https://github.com/username/repository.git  #add link if nothing present
git remote set-url origin git@github.com:username/repository.git  #change link, in this case to use ssh

# (1) edit your files + track/untrack/stage files
git status
git mv {oldfilename} {newfilename}  #rename a file. do this with git to properly be tracked!
git diff        #show changes
git add {file}  #add a file to staging for commit
git add .       #add all the files to staging for commit

# (2) commit changes
git commit -m "{comment}"  #finalizing commit for push, -a automatically stage modified and deleted files in this step
git log                    #show commit history

# (3) push changes
git branch -M main  #set the branch if needed
git push -u origin main  OR  git push --all

git pull
