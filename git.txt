# set up a local repository
mkdir /home/user1/git
#navigate to that directory
git clone https://github.com/username/repository.git
OR
git init  #initialize a local repository, creates all the necessary files in .git folder
#set your identity
git config user.name  AND  git config user.email  #shows what is being used
git config user.name {name}                 #set name
git config user.email {dummy-github-email}  #set email
#set the remote repository
git remote -v  #check the link to the remote repo
git remote add origin https://github.com/username/repository.git  #add link if nothing present
git remote set-url origin git@github.com:username/repository.git  #change link, in this case to use ssh

# use your best practices for generating and managing ssh keys
# GO TO https://github.com LOG IN and upload your public key
ssh -T git@github.com  #test the connection

file: ~/.ssh/config
  Host github-personal
    Hostname github.com
    User git
    IdentityFile ~/.ssh/{private-keyfile}
    IdentitiesOnly yes

# (1) edit your files
git mv {oldfilename} {newfilename}  #rename a file. do this with git to properly be tracked!
git rm {file}           #delete a file + add the deletion to staging
git rm -r {folder}      #delete a folder + add the deletion to staging
git rm --cached {file}  #removes + untracks the file but does not delete it
git diff                #show changes

# (2) staging for commit
# here you can review, adjust, remove changes
git status      #keep an eye on working files, set what files git tracks
git add {file}  #add a file to staging for commit. this can work: git add *.txt
git add --all   #add the entire project to staging  OR -A
git add .       #add all the files in current working directory and subdirectories to staging
git add *       #stages new and modified files but NOT deleted ones
git reset       #removes everything from staging

# (3) commit changes
git commit -m "{comment}"  #finalizing commit for push, -a automatically stage modified and deleted files in this step
git log                    #show commit history
git reset HEAD~            #undo the last commit set everything back to before previous stage

# (4) push changes
git branch -M main  #set the branch if needed
git push -u origin main  OR  git push --all

git pull
