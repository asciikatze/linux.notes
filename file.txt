-+-+-+-+-+-+ file manipulation +-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+

cat {file1} >> {file2}         #adds file1 to the end of file2

echo "{randomtext}" > file.txt   #replaces the text in the file
echo "{randomtext}" >> file.txt  #adds to the end of the file

-+-+-+-+-+-+ file permissions -+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
chown {user}:{group} {filename}      #set file ownership
chown -R {user}:{group} {directory}  #set folder ownership, -R apply to files within directory
chown -h {user}:{group} {linkfile}   #change the owner of a symbolic link

# file type d--------- :  regular file=-  directory=d  link=l
# octal values:  read=4  write=2  execute=1

  permission  octal  field
  -rwx------  700    User
  ----rwx---  070    Group
  -------rwx  007    Other

# when applying permissions to directories the permission bits have different meanings than on regular files.
# write bit allows the affected user to create, rename, or delete files within the directory, and modify the directory's attributes.
# read bit allows the affected user to list the files within the directory.
# execue bit allows the affected user to enter the directory and access files and directories inside.

chmod 755 {filename}              #set a file to be executable and readable by everyone but only writable by you
chmod +x {filename}               #set execute bit without changing other permissions, -x removes it
chmod -R u=rwx,g=,o= {directory}  #set permissions for folder and its contents

-+-+-+-+-+-+ about files +-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
file {file}   #information about file
stat {file}   #information about file, inode number and metadata

# files have: 1.file name, 2.inode/metadata, 3.file contents
# file name points to inode, inode points to file contents
# directory points to filenames and correspondign inodes

# hardlink: links to same inode
ln {sourcefile} {newsamefile}
# they have the same inode ergo they are the same file
# both files are the same no real file bullshit here
# has to be on the same device; devicenumber identifies the filesystem

# symlink: not the same file, different inode
ln -s {sourcefile} {newlinkfile}
# symlink points to a path to a file
# on symlinks use lstat for the linkfile information
# stat follows where the link follows through

# a directory has minimum 2 links to it
# the parent directory links to it and itself hardlinked to itself, the ./ hidden file in a directory is that
# if there are more links those are subdirectories linked to it by default ../
# the root directory's ./ and ../ are the same 
# user can not make hard links to a directory

du -hs {file}  #size of a file or folder
du -hs *       #size of the current working directory

diff {file1} {file2}  #compares the two files shows whats different
cmp {file1} {file2}   #compares byte by byte
wc {file}             #counts new lines, words and bytes in a file

-+-+-+-+-+-+ search -+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
# grep is for finding text in a file
grep "{sting}" {file}  #look for a string in a file, you can add multiple files
grep -r "{sting}" .    #look for a string in all the files in the current directory

# find is for finding files
find {directory} -iname a*              #search for files starting with a or A. case insensitive search
find {directory} -name a*               #search for files starting with a
find . -iname "{string}"                #USE THIS!
find {directory} -mtime -7              #search for files that are modified in the last 7 days
find {directory} -mmin -7               #search for files that are modified in the last 7 minutes
find {directory} -mtime -7 -size +10M   #search for files that are modified in the last 7 days and bigger than 10Mb, you can us '-' for smaller
find {directory} -size +10M | grep txt  #narrows the list to contain txt in the filenames
sudo find / -name {filename}            #search for a specific filename on the whole computer

#find in current directory all jpg and png files that are modified in the last year and copies it to the Pictures folder
find . \( -name '*.jpg' -o -name '*.png' \) -mtime -365 -exec cp {} /home/user1/Pictures \;
