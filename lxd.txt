# uses qemu for vms and lxc for containers
# cgroups for resource allocation and namespaces for isolation
# vms have their own kernel, containers share the kernel with the host machine
# lxd manages the vms lxc interacts with the vms
# lxd can be run nested under lxd
# incus is the community driven version of lxd. incusOS is a debian based linuxOS.
# help: https://www.youtube.com/playlist?list=PLDbRgZ0OOEpX_uqK_hAq98ltdc4DP-D34
# help: https://www.youtube.com/@LXD/videos
# help: https://documentation.ubuntu.com/lxd/latest/

# install and setup
snap install lxd
lxd init     #initial setup. pay attention to the pool and network configuration!
lxc version  #client-version=lxc  server-version=lxd

-+-+-+-+-+-+ create instances -+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
lxc remote list                      #available remote image servers
lxc image list images: debian        #lists all the debian images on the remote server called images
lxc image info images:{fingerprint}  #show more info about image; like usable alias
# the repository called images: https://images.lxd.canonical.com/
# difference between default and cloud images is that cloud images are work with cloudinit.
# there are container and vm images.
lxc image list  #show available images locally

lxc launch --vm {image} {vmname}              #--vm launches a virtual machine; whitout it lanches a conatainer
lxc launch --vm images:ubuntu/23.04 {vmname}  #you can specify the remote and the specific image you need

# create vm with custom ISO image
# the init creates the vm but does not launch it
lxc init --vm --empty --config=limits.cpu=3 --config=limits.memory=6GiB --device root,size=60GiB {vmname}
lxc config device add {vmname} iso disk source={file.iso}  #attach installation iso file to vm
lxc config device remove {vmname} iso                      #remove installation media
# if you want to install graphical vm see remote managment options

# make a windows vm
# help: https://blog.simos.info/how-to-run-a-windows-virtual-machine-on-incus-on-linux/
# help: https://www.youtube.com/watch?v=amslKipAjxo

lxc list                       #list of containers/vms
lxc list -c n,s,4,image.description:image  #you can edit what the list shows
lxc info {vmname}              #status. created/used times. disk cpu ram usage. ip. +available snapshots and more
lxc stop {vmname}              #stops the vm. stop  AND  start  AND  restart  are all valid
lxc delete {nvmame}            #you need to stop it first. you can add --force at the end
lxc copy {nvmame} {newnvmame}  #make a copy of the vm

-+-+-+-+-+-+ interact with instances +-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
lxc exec --help                 #help for exec command. --help helps with all commands!
lxc exec {vmname} -- {command}  #run a command against the vm
lxc exec {vmname} -- ls -lGa /home/user1  #list files on vm from home directory
lxc exec {vmname} -- bash       #hop in the vm terminal. can be bash or sh or whatever available on the vm

# help: https://documentation.ubuntu.com/lxd/latest/reference/manpages/lxc/file/#lxc-file-md
lxc file push {file} {vmname}/{pathtofile}  #copy file to vm. push ssh keys this way
lxc file pull {vmname}/{pathtofile} {file}  #copy from vm
lxc file edit {vmname}/{pathtofile}         #edit file on vm
lxc file delete {vmname}/{pathtofile}       #delete

# mount a directory from the vm to local system(uses SSHFS)
# useful when you manage the servers from a management computer
lxc file mount {vmname}/{pathtofolder}

# SSHSFTP through LXD daemon
# gives you username and password then you can use whatever program you like to connect
lxc file mount --listen=0.0.0.0:{port} {vmname}  #instead of 0.0.0.0 you can use localhost 127.0.0.1

lxc console {vmname}              #open the console for the vm
lxc start {vmname} --console=vga  #starts the vm with graphical window
lxc console {vmname} --type=vga   #connect graphical window. uses SPICE needs viewer program

-+-+-+-+-+-+ configure instances +-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
# default config is 1GiB memory, 1cpu core, 10GiB root hdd

# memory and cpu
lxc config set {vmname} limits.memory={value}  #resize memory vm needs to be stopped
lxc config set {vmname} limits.cpu={value}     #add/remove cpu cores vm can run during this. hot pluggable
# you can pin a specific core to a vm but that is not hot pluggable
# cpu will be in offline mode after you add one
lxc exec {vmname} -- lscpu --extended             #see which core is offline
lxc exec {vmname} -- chcpu --enable {corenumber}  #enable new core on vm
# example with different cpu andd memory limits:
lxc launch --vm --config limits.cpu=2 --config limits.memory=3GiB images:ubuntu/23.04 MyFirstVM

# root disk and data disk
lxc launch --vm --device root,size=25GiB images:ubuntu/23.04 MyFirstVM  #create vm with bigger root disk
# resize the vm root disk
lxc config device override {vmname} root size=50GiB  #reboot the vm or edit while it stopped
lxc exec {vmname} -- cfdisk  #resize the partition with the method you like
# create additional 10GiB data disk then attach(hot pluggable) it to a vm:
lxc storage volume create --type=block {poolname} {newdiskvolume} size=10GiB
lxc storage volume show {poolname} {diskvolume}  #informaton about the datadisk
lxc storage volume attach {poolname} {diskvolume} {vmname} {devicenameforvm}  #device name for vm is not manadtory
lxc storage volume delete {poolname} {diskvolume}  #detach from vm before deletion!
# the data disk can be on a different pool than the vm root disk

# set static ip for vm
lxc config device override {vmname} eth0 ipv4.address={static ip}  #stop before or restart the vm after

# edit configuration
lxc config show {vmname}             #show vm configuration
lxc config show --expanded {vmname}  #show vm configuration + device information network disk
# two ways to edit vm configuration:
lxc config edit {vmname}                 #edit the vms configuration file
lxc config set {vmname} {configurations changes here one by one}  #set more configs with lxc command

# help: https://documentation.ubuntu.com/lxd/stable-5.0/reference/instance_options/
lxc config set {vmname} boot.autostart=ture               #automatically boots when host boots
lxc config set {vmname} boot.autostart.priority={number}  #starting with the highest 
lxc config set {vmname} boot.autostart.delay=10           #add this to each vm for a 10 second sequenced boot

# tips: you can disable ethernet because ISO vms try to PXE boot by default when you install with them
lxc config device override {vmname} eth0 type=none
OR config edit {vmname}
devices:
  eth0:
    type: none
# restore the network interface
config edit {vmname}
devices:
  eth0:
    type: nic
    nictype: bridged
    parent: lxdbr0

lxc config get {vmname} {configuration}  #see one specific configuration setting

-+-+-+-+-+-+ configure profiles -+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
# default profile applied when creating a vm if not instructed otherwise
# with profiles you can pre set: root disk size, target storage pool, etc..

lxc profile list
lxc profile create {profilename}
lxc profile show {profilename}

lxc profile set {profilename} {parameters and values}  #edit profile by command
OR
lxc profile edit {profilename}  #edit by text editor

lxc profile add {vmname} {profilename}  #you can append more than one profile to a vm
# profile settings conflicts settled by profile ASSIGN ORDER: the LAST profiles SETTING APPLY!
lxc profile assign {vmname} {profilename},{profilename}  #you can change profile order with this?
# when you edit a profile that is in use the new CONFIG APPLIES to the RUNNING VMs that use that profile!
# most cases reboot or manual configuration still needed when applied to vms in use!

lxc launch --vm --profile default --profile {profilename} {image} {vmname}  #applied 2 profiles at creation

-+-+-+-+-+-+ snapshots +-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
lxc snapshot {vmname} {snapshotname}  #takes a snapshot. snapshotname is not required
lxc info {vmname}                     #available snapshots
lxc restore {vmname} {snapshotname}
# zfs pool LIMITATION ! you can onyl restore the latest snapshot
# workaround 1 delete snapshots until what you want is the latest
# workaround 2 create a new vm from an older snapshot
lxc copy {vmname}/{snapshotname} {newvmname}  #create a new vm from snapshot
lxc delete {vmname}/{snapshotname}            #delete a snapshot

# help: https://documentation.ubuntu.com/lxd/latest/howto/instances_backup/#instances-snapshots
# schedule periodic vm snapshots
# automatically snapshot a stopped vm
# set default snapshot naming pattern

# set expiration time period for snapshots
file: lxc config edit {vmname}
config:
  snapshots.exiry: {time}  #time format: 1M 2H 3d 4w 5m 6y
OR lxc config set {vmname} snapshots.exiry={time}

-+-+-+-+-+-+ backups +-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
# backups are simpler to use for migrating
# help: https://documentation.ubuntu.com/lxd/latest/howto/instances_backup/#use-export-files-for-instance-backup

# use exprt command while the vm is shut down
lxc export {vmname} {filename} --compression="gzip -1"  #filename and compression not required
# level of compression: 1 fast with less compression - 9 slow with more compression
# makes the file on the host filesystem where you issued the command

lxc import {filename} {newvmname}  #newvmname is not required
lxc import {filename} {newvmname} --storage={poolname}
# after import manage the hostname and fingerprints and other identifiers because those will be the same
# you made the setup on your laptop then you push it to your lxd server:
lxc import {targetremotelxdserver} {filename} {newvmname}

-+-+-+-+-+-+ storage pools +-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
# storage pool types: Dir Btrfs Lvm Zfs-default Ceph-remote
# simple Dir: you can just point to an empty directory on host system disk
# location for default dir type pool: /var/snap/lxd/common/lxd/storage-pools/default/
lxc storage create {poolname} {pooltype}  #create a storage pool

# example for Dir type pool:
mkdir -p /home/lxdadm/lxdpool-2                                 #make a directory for your dir pool
lxc storage create lxdpool-2 dir source=/home/lxdadm/lxdpool-2  #creates a dir type pool in a folder

lxc storage list                          #shows available storage pools
lxc storage volume list {poolname}        #shows contents of a storage pool. image vm datadisk
lxc launch --storage={poolname} --vm {image} {vmname}  #create a vm on a specific storage pool

-+-+-+-+-+-+ virtual network +-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
# default bridge network 10.x.x.x/24 for the vms. host connects to it through lxdbr0 virtual interface
# vm can route through the host. other machines from outside can not route to the vm
lxc network list  #available networks on host

# bridged network forward one of the hosts port to vm port
# help: https://documentation.ubuntu.com/lxd/latest/howto/network_forwards/
lxc network forward create {lxdbr0} {hostip}
lxc network forward port add {lxdbr0} {hostip} {tcp|udp} {hostport} {vmip} {vmport}

# forward a secondary host ip to a vm
# for a permanently solution edit the netplan *config.yaml file and add another ip there!
# temporarly configure secondary address for host(del for delete)
sudo ip address add {new2ndhostip/24} dev {eth0}
lxc network forward create {lxdbr0} {new2ndhostip} target_address={vmip}  #ip forward rule

lxc network forward list {lxdbr0}  #list network forwards for interface
lxc network forward show {lxdbr0} {hostip}
lxc network forward edit {lxdbr0} {hostip}
lxc network forward delete {lxdbr0} {hostip}

-+-+-+-+-+-+ firewall on host -+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
# configure the host firewall ufw. applies to managed bridge networks only !
# help: https://documentation.ubuntu.com/lxd/latest/howto/network_bridge_firewalld/

# disable lxd firewall rules
lxc network set lxdbr0 ipv4.firewall false
lxc network set lxdbr0 ipv6.firewall false

# quick and dirty
sudo ufw allow in on lxdbr0
sudo ufw route allow in on lxdbr0
sudo ufw route allow out on lxdbr0

# more restrictive firewall rules
# limit access from the guests to the host to only DHCP and DNS and allow all outbound connections.
# allow the guest to get an IP from the LXD host
sudo ufw allow in on lxdbr0 to any port 67 proto udp   #ipv4 dhcp
sudo ufw allow in on lxdbr0 to any port 547 proto udp  #ipv6 dhcp
# allow the guest to resolve host names from the LXD host
sudo ufw allow in on lxdbr0 to any port 53
# allow the guest to have access to outbound connections
lxc network get lxdbr0 ipv4.address  #show the lxd network
lxc network get lxdbr0 ipv6.address
sudo ufw route allow in on lxdbr0 from {lxd network}

# allow traffic from outside. after you set up the lxd forwarding rules
sudo ufw route allow out on {lxdbr0} from {192.168.x.x/24}
